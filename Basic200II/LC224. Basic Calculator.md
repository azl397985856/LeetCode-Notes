#### 代码 JavaScript

```JavaScript
/**
 * @param {string} s
 * @return {number}
 */
var calculate = function(s) {
    let ans = 0
    const num = []
    const opr = []
    let sign = 1

    for(let i = 0; i < s.length; i++){
        if(s[i] == ' ')continue;

        if(s[i] == '+' || s[i] == '-'){
            sign = s[i] == '+' ? 1 : -1
        }else if(s[i] >= '0' && s[i] <= '9'){
            let num = Number(s[i])
            while(s[i + 1] >= '0' && s[i + 1] <= '9' && i < s.length){
                i++
                num = num * 10 + Number(s[i])
            }
            ans += num * sign
        }else if(s[i] == '('){
            num.push(ans)
            opr.push(sign)
            ans = 0
            sign = 1
        }else if(s[i] == ')'){
            ans = num.pop() + ans * opr.pop()
        }
    }

    return ans
};

```

#### 复杂度分析
时间复杂度：O(N) </br>
空间复杂度：O(N)
