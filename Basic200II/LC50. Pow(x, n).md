### 思路 二分

快速幂

#### 代码 JavaScript

```JavaScript
var myPow = function(x, n) {
    if(n == 1) return x
    if(n == 0) return 1
    if(n < 0){
        return 1/quickPow(x, -n)
    }else{
        return quickPow(x, n)
    }

};

var quickPow = function(x, n){
    if(n == 1) return x
    if(n == 0) return 1
    const halfn = Math.floor(n / 2)
    const res = quickPow(x, halfn)
    if(n % 2 == 0){
        return res * res
    }else{
        return res * res * x
    }
}

```
