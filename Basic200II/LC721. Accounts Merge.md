### 思路 并查集

#### 代码 JavaScript

```JavaScript
/**
 * @param {string[][]} accounts
 * @return {string[][]}
 */
var accountsMerge = function(accounts) {
    const n = accounts.length
    const memo = new Map // email -> index
    const uf = new UnionFind(n)
    for(let i = 0; i < n; i++){
        uf.make(i, accounts[i])
        for(let j = 1; j < accounts[i].length; j++){
            const email = accounts[i][j]
            if(!memo.has(email)){
                memo.set(email, i)
            }else{
                const bro = memo.get(email)
                uf.union(bro, i)
            }
        }
    }

    const res = []
    for(let i = 0; i < n; i++){
        if(uf.isParent(i)){
            // console.log('acc', accounts[i])
            let temp = [accounts[i][0]]
            temp = temp.concat(uf.emails(i))
            res.push(temp)
        }
    }
    
return res

};

class UnionFind{
    constructor(n){
        this.parent =  new Array(n)
        this.memo = new Map
        // for(let i = 0; i < n; i++){
        //     this.parent.push(i)
        //     this.memo.set(i, [])
        // }
    }

    make(n, emails){
        let copy = emails.slice()
        copy.shift()
        this.parent[n] = n
        this.memo.set(n, copy)
    }

    find(x){
        while(x != this.parent[x]){
            x = this.parent[x]
        }

        return x
    }

    union(a,b){
        let x = this.find(a)
        let y = this.find(b)
        if(x == y) return false

        this.parent[y] = x
    }

    isParent(a){
        return this.parent[a] == a
    }

    emails(a){
        const res = new Set
        for(let i = 0; i < this.parent.length; i++){
            if(this.find(i) == a){
                for(const e of this.memo.get(i)){
                    res.add(e)
                }
            }
        }
        // console.log(a, res)
        return [...res].sort()
    }
}
```
