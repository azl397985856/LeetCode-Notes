var backtrack = function(list, tempList, nums, remain, start){
  if(remain < 0) return;
  else if (remain == 0) return list.push([...tempList]);
  for(let i = start; i < nums.length; i++){
    tempList.push(nums[i]);
    backtrack(list,tempList, nums, remain - nums[i], i);
    tempList.pop();
  }
}

const list = [];

return list;
};
